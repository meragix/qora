      appBar: AppBar(

            success: (user, updatedAt) => UserDetailView(user: user),
            failure: (err, _, __) => Center(child: Text('Error: $err')),
          );
        },
      ),
    );
  }
}

class UserDetailView extends StatelessWidget {
