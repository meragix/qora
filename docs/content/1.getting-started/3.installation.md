## Flutter

### 1. Add the dependencies

Add `flutter_qora` and `qora` to your `pubspec.yaml`:

```yaml [pubspec.yaml]
dependencies:
  flutter:
    sdk: flutter
  qora: ^1.0.0
  flutter_qora: ^1.0.0
```

### 2. Install

```bash
flutter pub get
```

### 3. Basic setup

Create a `QoraClient` and wrap your app with `QoraScope`:

```dart [main.dart]
import 'package:flutter/material.dart';
import 'package:flutter_qora/flutter_qora.dart';

void main() {
  final client = QoraClient(
    config: QoraClientConfig(
      defaultOptions: QoraOptions(
        staleTime: Duration(minutes: 5),
        cacheTime: Duration(minutes: 10),
        retryCount: 3,
      ),
      debugMode: true,
    ),
  );

  runApp(
    QoraScope(
      client: client,
      child: const MyApp(),
    ),
  );
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(home: HomeScreen());
  }
}
```

:::tip
Set `debugMode: kDebugMode` in production builds so debug logs are stripped automatically.
:::

You're ready. Head to the [Quick Start](./quick-start.md) to build your first screen.

## Pure Dart (no Flutter)

If you are building a CLI tool, backend service, or shared package without Flutter, use the core package directly:

```yaml [pubspec.yaml]
dependencies:
  qora: ^1.0.0
```

```bash
dart pub get
```

### Usage

```dart [main.dart]
import 'package:qora/qora.dart';

void main() async {
  final client = QoraClient(
    config: const QoraClientConfig(debugMode: true),
  );

  // One-shot fetch with caching and retry
  final user = await client.fetchQuery<User>(
    key: ['users', 1],
    fetcher: () => api.getUser(1),
  );

  print('User: ${user.name}');

  // Reactive stream — emits every state transition
  client.watchQuery<User>(
    key: ['users', 1],
    fetcher: () => api.getUser(1),
  ).listen((state) {
    switch (state) {
      case Success(:final data):  print('Updated: ${data.name}');
      case Failure(:final error): print('Error: $error');
      default: {}
    }
  });

  client.dispose();
}
```

## Optional: Lifecycle and Connectivity Managers

For automatic refetch-on-resume and refetch-on-reconnect, add the Flutter managers to `QoraScope`:

```dart [main.dart]
import 'package:flutter_qora/flutter_qora.dart';

void main() {
  final client = QoraClient();

  runApp(
    QoraScope(
      client: client,
      lifecycleManager: FlutterLifecycleManager(
        qoraClient: client,
        refetchInterval: const Duration(seconds: 30),
      ),
      connectivityManager: FlutterConnectivityManager(
        qoraClient: client,
      ),
      child: const MyApp(),
    ),
  );
}
```

`FlutterConnectivityManager` requires the [`connectivity_plus`](https://pub.dev/packages/connectivity_plus) package:

```yaml [pubspec.yaml]
dependencies:
  connectivity_plus: ^6.0.0
```

## Verify the Installation

```dart [verify.dart]
import 'package:qora/qora.dart';

void main() {
  final client = QoraClient();
  print('Qora installed successfully.');
  client.dispose();
}
```

```bash
dart run verify.dart
# Qora installed successfully.
```
